Git Essential Commands

# Create feature branch
git checkout -b feature/nav-fix

# Commit with message
git commit -m "feat: resolve navbar overflow"

# Push branch to remote
git push -u origin feature/nav-fix

# Tag release
git tag -a v1.0.0 -m "v1.0.0 release"
git push origin release --tags

CSS Grid & Flexbox Snippets

/* Centered flex container */
.center {
  display: flex;
  align-items: center;
  justify-content: center;
}

/* Responsive grid */
.grid {
  display: grid;
  grid-template-columns: repeat(auto-fit, minmax(280px, 1fr));
  gap: 1.5rem;
}
